約数列挙

2019/09/25

概要

約数を列挙する

計算量

* O(√n)

実装例

python 
      
        def divi(n):
          res = []
          for i in range(1, int(n**0.5)+1):
            if n % i == 0:
              res.append(i)
              if i != n // i:
                res.append(n // i)
          return res
C++ 
      
        template<typename T>
        vector<T> divi(T n) {
          vector<T> res;
          for (T i = 1; i * i <= n; ++i) {
            if (n % i == 0) {
              res.push_back(i);
              if (i * i != n) res.push_back(n / i);
            }
          }
          sort(begin(res), end(res));
          return res;
        }
